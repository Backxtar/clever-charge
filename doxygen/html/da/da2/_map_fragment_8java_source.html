185 private boolean setPoints(final double lat, final double lon, boolean onMapClick) {
186 ArrayList<ChargingStation> stations_near = StationManager.sort(lat, lon);
187 BitmapDescriptor bitMap = bitMapFromPng(getActivity(), R.drawable.ic_marker);
188 BitmapDescriptor bitMapFav = bitMapFromPng(getActivity(), R.drawable.ic_marker_fav);
189
190 if (stations_near.isEmpty()) {
191 MessageService msgService = new MessageService(getActivity(), getResources().getString(R.string.no_stations_available), Gravity.CENTER, true);
192 msgService.sendToast();
193 return false;
194 }
195 stations_near.forEach(station -> {
196 final double distance;
197 if (!onMapClick) distance = station.getDistance();
198 else distance = station.getDistance(lat, lon);
199 MarkerOptions markerOptions = new MarkerOptions();
200 markerOptions.position(new LatLng(station.getLat(), station.getLon()));
201
202 if (UserManager.getApi_data().getDefect_stations_map().containsKey(station.getId()))
203 markerOptions.icon(BitmapDescriptorFactory.defaultMarker());
204 else if (UserManager.getApi_data().getFavorites().contains(station.getId()) &&
205 !UserManager.getApi_data().getDefect_stations_map().containsKey(station.getId()))
206 markerOptions.icon(bitMapFav);
207 else markerOptions.icon(bitMap);
208
209 markerOptions.title(station.getOperator() + " | "
210 + String.format(Locale.getDefault(),"%.2f", distance)
211 + "km | " + getResources().getString(R.string.more) + "...");
212 Marker marker = googleMap.addMarker(markerOptions);
213 marker.setTag(station);
214 });
215 return true;
216 }

224 public static BitmapDescriptor bitMapFromPng(Context context, int drawableID) {
225 Drawable drawable = ContextCompat.getDrawable(context, drawableID);
226 drawable.setBounds(0, 0, drawable.getIntrinsicWidth(), drawable.getIntrinsicHeight());
227 Bitmap bitmap = Bitmap.createBitmap(drawable.getIntrinsicWidth(), drawable.getIntrinsicHeight(), Bitmap.Config.ARGB_8888);
228 Canvas canvas = new Canvas(bitmap);
229 drawable.draw(canvas);
230 return BitmapDescriptorFactory.fromBitmap(bitmap);
231 }
