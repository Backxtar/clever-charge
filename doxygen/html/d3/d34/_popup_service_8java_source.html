270 if (fav != null && nearby != null) {

278 stations = StationManager.sortByFavAndLocal();

280 stations = StationManager.sort(
281 UserManager.getMyPosition().latitude,
282 UserManager.getMyPosition().longitude);
283 else if (!nearby.isChecked() && fav.isChecked())
284 stations = StationManager.getFavorites();
285 else stations = StationManager.getStation_list();
286 }
287 else if (fav == null && nearby != null) {
288 if (nearby.isChecked())
289 stations = StationManager.getLocalDefects(
290 UserManager.getMyPosition().latitude,
291 UserManager.getMyPosition().longitude);
292 else stations = StationManager.getDefects();
293 }
294 else return;
295 stations.sort(StationManager.getSortByDistance());
