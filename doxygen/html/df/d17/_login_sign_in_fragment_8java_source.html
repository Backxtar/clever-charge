122 private void callApi(Intent intent, String user, String passwd, CheckBox remember_me) {
123 CompletableFuture.supplyAsync(() -> DownloadService.getResponse("login",
124 new String[]{"username=" + user, "userpasswd=" + passwd}))
125 .whenComplete((apiResponse, throwable) -> {
126 if (throwable != null ||
127 apiResponse.getResponseCode() != 1 ||
128 apiResponse.getDefect_stations_map() == null ||
129 apiResponse.getFavorites() == null) {
130 MessageService msgService = new MessageService(getActivity(), getResources().getString(R.string.login_failed), Gravity.TOP, true);
131 msgService.sendToast();
132 return;
133 }
134 UserManager.setApi_data(apiResponse);
135
136 if (remember_me.isChecked()) {
137 SharedPreferences sharedPref;
138 if (getActivity() != null) {
139 sharedPref = getActivity().getSharedPreferences("MySet", Context.MODE_PRIVATE);
140 SharedPreferences.Editor editor = sharedPref.edit()
141 .putString("USER_NAME", user)
142 .putString("USER_PASSWD", passwd)
143 .putBoolean("REMEMBER_ME", true)
144 .putInt("RANGE", 10);
145 editor.apply();
146 }
