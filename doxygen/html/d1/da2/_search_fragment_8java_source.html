110 private void initSwipeRefresh(View view) {

112 refreshLayout.setColorSchemeResources(R.color.primary);
113 refreshLayout.setOnRefreshListener(() -> {

115 .whenComplete((apiResponse, throwable) -> {
116 if (throwable != null || apiResponse.getResponseCode() != 1) {
117 MessageService msgService = new MessageService(getActivity(), getResources().getString(R.string.cant_load_defect_stations), Gravity.TOP, true);
118 msgService.sendToast();
119 } else UserManager.getApi_data().setDefect_stations_map(apiResponse.getDefect_stations_map());
120 });

122 CompletableFuture<ArrayList<ChargingStation>> futureStationResponse = CompletableFuture.supplyAsync(DownloadService::getStations)
123 .whenComplete((stations, throwable) -> {
124 if (throwable != null || stations == null) {
125 MessageService msgService = new MessageService(getActivity(), getResources().getString(R.string.cant_get_station_list), Gravity.TOP, true);
126 msgService.sendToast();
127 } else StationManager.setStation_list(stations);
128 });
129
130 CompletableFuture.allOf(futureResponseDef, futureStationResponse)
131 .whenComplete((unused, throwable) -> {
132 MessageService msgService;
133
134 if (throwable != null) {
135 msgService = new MessageService(getActivity(), getResources().getString(R.string.error_while_getting_data), Gravity.TOP, true);
136 msgService.sendToast();
137 return;
138 }
139 msgService = new MessageService(getActivity(), getResources().getString(R.string.update_successful), Gravity.TOP, false);
140 msgService.sendToast();
141 if (getActivity() != null) {
142 getActivity().runOnUiThread(() -> {
143 updateAdapter();
144 refreshLayout.setRefreshing(false);
145 });
146 }

210 private ArrayList<ChargingStation> filterList(Editable s) {
211 return StationManager.getStation_list().stream()
212 .filter(station -> station.getOperator().toLowerCase().contains(s.toString().toLowerCase()) ||
213 station.getStreet().toLowerCase().contains(s.toString().toLowerCase()) ||
214 station.getNumber().toLowerCase().contains(s.toString().toLowerCase()) ||
215 Integer.toString(station.getPostal_code()).toLowerCase().contains(s.toString().toLowerCase()) ||
216 station.getLocation().toLowerCase().contains(s.toString().toLowerCase()) ||
217 StationManager.convertEnum(getActivity(), station.getModule_type()).toLowerCase().contains(s.toString().toLowerCase()))
218 .sorted(StationManager.getSortByDistance()).collect(Collectors.toCollection(ArrayList<ChargingStation>::new));
219 }
